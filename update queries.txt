-- 1. Controllable Head Count (ID = 11)
-- ============================================================
UPDATE "T1GDTL_PHY"."NRFP_FINANCIALS_REPORTINGMASTER"
SET QUERY = '
-- ============================================================
-- WIDGET  : Controllable Headcount - Numeric Point
-- REPORT  : CIB Dashboard
-- SOURCE  : T1NFRP_PHY.CV_UP_NFRP_ALLDATA  (primary)
--           T1NFRP_PHY.TBL_NFRP_DATE_CONFIG (period resolution)
-- FILTERS : M_ACCOUNT_0 = ''Headcount''
--           M_ACCOUNT_1 = ''Controllable FTE''
--           T_REPORTING = ''FPNA''
-- OUTPUT  : 14 columns — 7 measures × CFX + RFX separately
-- ============================================================
-- PERIOD MAPPING:
--   ADD16 = ''0''              → CURRENT_YTD_PERIOD
--   ADD17 = ''-1'', ADD19=''CY'' → CY_PQ_PERIOD
--   ADD17 = ''-5'', ADD19=''PY'' → PY_PQ_PERIOD
--   ADD6  = ''PY_YTD''         → YTD_PY_PERIOD
--   ADD19 = ''CY'', MONTH_ABR=''FY'' → FY_OUTLOOK_PERIOD
-- ============================================================

WITH DATA_ANCHOR AS (
    SELECT
        MAX(A.E_PERIOD_DATE)                    AS ACTUALS_YTD_DATE
    FROM T1NFRP_PHY."CV_UP_NFRP_ALLDATA" A
    WHERE A.M_ACCOUNT_0  = ''Headcount''
    AND   A.M_ACCOUNT_1  = ''Controllable FTE''
    AND   A.T_REPORTING  = ''FPNA''
    AND   A.DATA_TYPE    = ''Actuals''
    AND   A.T_VERSION    = ''YTD''
    AND   A.CFX          != 0
    AND EXISTS (
        SELECT 1
        FROM T1NFRP_PHY."TBL_NFRP_DATE_CONFIG" D
        WHERE D.FILTER_DATE   = TO_VARCHAR(A.E_PERIOD_DATE, ''YYYY-MM-DD'')
    )
),

PERIODS AS (
    SELECT
        MAX(CASE WHEN D.ADD16 = ''0''
            THEN D.E_PERIOD_DATE END)           AS CURRENT_YTD_PERIOD,
        MAX(CASE WHEN D.ADD17 = ''-1''
            AND D.ADD19 = ''CY''
            THEN D.E_PERIOD_DATE END)           AS CY_PQ_PERIOD,
        MAX(CASE WHEN D.ADD17 = ''-5''
            AND D.ADD19 = ''PY''
            THEN D.E_PERIOD_DATE END)           AS PY_PQ_PERIOD,
        MAX(CASE WHEN D.ADD6  = ''PY_YTD''
            THEN D.E_PERIOD_DATE END)           AS YTD_PY_PERIOD
    FROM T1NFRP_PHY."TBL_NFRP_DATE_CONFIG" D
    CROSS JOIN DATA_ANCHOR A
    WHERE D.FILTER_DATE = TO_VARCHAR(A.ACTUALS_YTD_DATE, ''YYYY-MM-DD'')
),

FY_OUTLOOK_ANCHOR AS (
    SELECT
        MAX(E_PERIOD_DATE)                      AS FY_OUTLOOK_PERIOD
    FROM T1NFRP_PHY."CV_UP_NFRP_ALLDATA"
    WHERE M_ACCOUNT_0   = ''Headcount''
    AND   M_ACCOUNT_1   = ''Controllable FTE''
    AND   T_REPORTING   = ''FPNA''
    AND   DATA_TYPE     = ''Outlook_PM''
    AND   T_VERSION     = ''YTD''
    AND   MONTH_ABR     = ''FY''
),

-- ✅ Pre-resolve all periods into one row
PERIOD_SET AS (
    SELECT
        P.CURRENT_YTD_PERIOD,
        P.CY_PQ_PERIOD,
        P.PY_PQ_PERIOD,
        P.YTD_PY_PERIOD,
        OA.FY_OUTLOOK_PERIOD
    FROM PERIODS             P
    CROSS JOIN FY_OUTLOOK_ANCHOR OA
),

-- ✅ Single scan — CFX and RFX computed separately in same pass
ALL_MEASURES AS (
    SELECT
        -- ══════════════════════════════════════════════
        -- CFX COLUMNS
        -- ══════════════════════════════════════════════

        -- ── CFX: YTD Actuals ──────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''YTD''
            AND  A.E_PERIOD_DATE = PS.CURRENT_YTD_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS YTD_ACTUALS_CFX,

        -- ── CFX: YTDB Budget ──────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Budget''
            AND  A.T_VERSION     = ''YTD''
            AND  A.E_PERIOD_DATE = PS.CURRENT_YTD_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS YTDB_CFX,

        -- ── CFX: CY PQ Actuals ────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''QTD''
            AND  A.E_PERIOD_DATE = PS.CY_PQ_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS CY_PQ_ACTUALS_CFX,

        -- ── CFX: FY Outlook ───────────────────────────
        -- Outlook_PM always RFX — CFX kept for reference
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Outlook_PM''
            AND  A.T_VERSION     = ''YTD''
            AND  A.MONTH_ABR     = ''FY''
            AND  A.E_PERIOD_DATE = PS.FY_OUTLOOK_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS FY_OUTLOOK_CFX,

        -- ── CFX: YTD Actuals PY ───────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''MTD''
            AND  A.E_PERIOD_DATE = PS.YTD_PY_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS YTD_ACTUALS_PY_CFX,

        -- ── CFX: PY PQ Actuals ────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''QTD''
            AND  A.E_PERIOD_DATE = PS.PY_PQ_PERIOD
            THEN A.CFX ELSE 0
        END)                                    AS PY_PQ_ACTUALS_CFX,

        -- ══════════════════════════════════════════════
        -- RFX COLUMNS
        -- ══════════════════════════════════════════════

        -- ── RFX: YTD Actuals ──────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''YTD''
            AND  A.E_PERIOD_DATE = PS.CURRENT_YTD_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS YTD_ACTUALS_RFX,

        -- ── RFX: YTDB Budget ──────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Budget''
            AND  A.T_VERSION     = ''YTD''
            AND  A.E_PERIOD_DATE = PS.CURRENT_YTD_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS YTDB_RFX,

        -- ── RFX: CY PQ Actuals ────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''QTD''
            AND  A.E_PERIOD_DATE = PS.CY_PQ_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS CY_PQ_ACTUALS_RFX,

        -- ── RFX: FY Outlook ───────────────────────────
        -- Outlook_PM always RFX
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Outlook_PM''
            AND  A.T_VERSION     = ''YTD''
            AND  A.MONTH_ABR     = ''FY''
            AND  A.E_PERIOD_DATE = PS.FY_OUTLOOK_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS FY_OUTLOOK_RFX,

        -- ── RFX: YTD Actuals PY ───────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''MTD''
            AND  A.E_PERIOD_DATE = PS.YTD_PY_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS YTD_ACTUALS_PY_RFX,

        -- ── RFX: PY PQ Actuals ────────────────────────
        SUM(CASE
            WHEN A.DATA_TYPE     = ''Actuals''
            AND  A.T_VERSION     = ''QTD''
            AND  A.E_PERIOD_DATE = PS.PY_PQ_PERIOD
            THEN A.RFX ELSE 0
        END)                                    AS PY_PQ_ACTUALS_RFX

    FROM T1NFRP_PHY."CV_UP_NFRP_ALLDATA" A
    CROSS JOIN PERIOD_SET PS
    WHERE A.M_ACCOUNT_0  = ''Headcount''
    AND   A.M_ACCOUNT_1  = ''Controllable FTE''
    AND   A.T_REPORTING  = ''FPNA''
    AND   A.E_PERIOD_DATE IN (
        PS.CURRENT_YTD_PERIOD,
        PS.CY_PQ_PERIOD,
        PS.PY_PQ_PERIOD,
        PS.YTD_PY_PERIOD,
        PS.FY_OUTLOOK_PERIOD
    )
)

SELECT
    -- ══════════════════════════════════════════════════════
    -- CFX: 7 columns
    -- ══════════════════════════════════════════════════════
    M.YTD_ACTUALS_CFX,

    -- Vs Budget: YTD Actuals – YTDB (absolute difference per spec)
    (M.YTD_ACTUALS_CFX - M.YTDB_CFX)               AS VS_BUDGET_CFX,

    M.CY_PQ_ACTUALS_CFX,
    M.FY_OUTLOOK_CFX,

    -- PY PQ Actuals YoY% : Compare(CY_PQ TO PY_PQ) as %
    -- Formula: (CY_PQ - PY_PQ) * 100 / ABS(PY_PQ)
    CASE WHEN M.PY_PQ_ACTUALS_CFX != 0
        THEN ROUND(
            (M.CY_PQ_ACTUALS_CFX - M.PY_PQ_ACTUALS_CFX)
            * 100.0 / ABS(M.PY_PQ_ACTUALS_CFX), 2)
    END                                             AS VAR_PY_PQ_YOY_PCT_CFX,

    -- YTD Actual PY YoY% : Compare(YTD Actuals TO YTD Actual PY) as %
    -- Formula: (YTD - YTD_PY) * 100 / ABS(YTD_PY)
    CASE WHEN M.YTD_ACTUALS_PY_CFX != 0
        THEN ROUND(
            (M.YTD_ACTUALS_CFX - M.YTD_ACTUALS_PY_CFX)
            * 100.0 / ABS(M.YTD_ACTUALS_PY_CFX), 2)
    END                                             AS VAR_YTD_PY_YOY_PCT_CFX,

    -- FY Outlook YoY% : Compare(FY Outlook TO YTD Actual PY) as %
    -- Formula: (FY_OUTLOOK - YTD_PY) * 100 / ABS(YTD_PY)
    CASE WHEN M.YTD_ACTUALS_PY_CFX != 0
        THEN ROUND(
            (M.FY_OUTLOOK_CFX - M.YTD_ACTUALS_PY_CFX)
            * 100.0 / ABS(M.YTD_ACTUALS_PY_CFX), 2)
    END                                             AS VAR_FY_VS_PY_YOY_PCT_CFX,

    -- ══════════════════════════════════════════════════════
    -- RFX: 7 columns
    -- ══════════════════════════════════════════════════════
    M.YTD_ACTUALS_RFX,

    -- Vs Budget: YTD Actuals – YTDB (absolute difference per spec)
    (M.YTD_ACTUALS_RFX - M.YTDB_RFX)               AS VS_BUDGET_RFX,

    M.CY_PQ_ACTUALS_RFX,
    M.FY_OUTLOOK_RFX,

    -- PY PQ Actuals YoY% : Compare(CY_PQ TO PY_PQ) as %
    CASE WHEN M.PY_PQ_ACTUALS_RFX != 0
        THEN ROUND(
            (M.CY_PQ_ACTUALS_RFX - M.PY_PQ_ACTUALS_RFX)
            * 100.0 / ABS(M.PY_PQ_ACTUALS_RFX), 2)
    END                                             AS VAR_PY_PQ_YOY_PCT_RFX,

    -- YTD Actual PY YoY% : Compare(YTD Actuals TO YTD Actual PY) as %
    CASE WHEN M.YTD_ACTUALS_PY_RFX != 0
        THEN ROUND(
            (M.YTD_ACTUALS_RFX - M.YTD_ACTUALS_PY_RFX)
            * 100.0 / ABS(M.YTD_ACTUALS_PY_RFX), 2)
    END                                             AS VAR_YTD_PY_YOY_PCT_RFX,

    -- FY Outlook YoY% : Compare(FY Outlook TO YTD Actual PY) as %
    CASE WHEN M.YTD_ACTUALS_PY_RFX != 0
        THEN ROUND(
            (M.FY_OUTLOOK_RFX - M.YTD_ACTUALS_PY_RFX)
            * 100.0 / ABS(M.YTD_ACTUALS_PY_RFX), 2)
    END                                             AS VAR_FY_VS_PY_YOY_PCT_RFX

FROM PERIOD_SET          PS
CROSS JOIN ALL_MEASURES  M;
'
WHERE ID = 11;
